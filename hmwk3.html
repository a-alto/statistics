RSA cryptosystem

What is RSA in cryptography?
RSA is a widely used public-key cryptosystem that enables secure data transmission, encryption, and digital signatures.

How does RSA work?
RSA is based on the mathematical properties of large prime numbers and modular arithmetic. Its security relies on the practical difficulty of factoring the product of two large prime numbers.

Main steps of RSA:

  • Key Generation:
    • Select two large prime numbers, \( p \) and \( q \).
    • Compute their product \( n = pq \); \( n \) is part of the public key and modulus for encryption/decryption.
    • Calculate Euler's totient: \( \varphi(n) = (p-1)(q-1) \).
    • Choose an integer \( e \) such that \( 1 < e < \varphi(n) \) and \( \gcd(e, \varphi(n)) = 1 \); \( e \) is the public exponent.
    • Compute the private exponent \( d \) as the modular inverse of \( e \) modulo \( \varphi(n) \):
      \( d \equiv e^{-1} \pmod{\varphi(n)} \)
    • Public key: \( (n, e) \)
      Private key: \( d \) (and in practice, \( p \) and \( q \) are also kept secret)
  • Encryption:
    To encrypt a message (as a number) \( m \) with the public key:
    \( c \equiv m^e \pmod{n} \)
  • Decryption:
    Use the private key \( d \) to decrypt ciphertext \( c \):
    \( m \equiv c^d \pmod{n} \)
  • Digital Signatures:
    • Sign a message by computing \( s \equiv m^d \pmod{n} \) using the private key.
    • Verification is done by checking \( m \equiv s^e \pmod{n} \) with the public key.
Summary:
RSA is fundamental to modern secure communications, providing both confidentiality (encryption) and authenticity (digital signatures) based on well-understood mathematical operations.

Breaking RSA




Here is the JavaScript code used to encrypt using RSA and break it (for small primes p and q): 
                            JavaScript
                            // Helper: check if n is prime (small numbers)
                            function isPrime(n) {
                                if (n < 2) return false;
                                for (let i = 2; i * i <= n; ++i)
                                    if (n % i === 0) return false;
                                return true;
                            }
                            // Greatest common divisor
                            function gcd(a, b) {
                                while (b !== 0) { let t = b; b = a % b; a = t; }
                                return a;
                            }
                            // Modular inverse of e mod phi
                            function modinv(e, phi) {
                                let [t, newt] = [0, 1], [r, newr] = [phi, e];
                                while (newr !== 0) {
                                    let q = Math.floor(r / newr);
                                    [t, newt] = [newt, t - q * newt];
                                    [r, newr] = [newr, r - q * newr];
                                }
                                if (r > 1) return null;
                                if (t < 0) t += phi;
                                return t;
                            }
                            // Modular exponentiation
                            function modpow(base, exp, mod) {
                                let result = 1;
                                base = base % mod;
                                while (exp > 0) {
                                    if (exp % 2 === 1) result = (result * base) % mod;
                                    exp = Math.floor(exp / 2);
                                    base = (base * base) % mod;
                                }
                                return result;
                            }
                            // String to UTF-16 char codes (for demo; only latin/ASCII works safely)
                            function strToCodes(s) {
                                return Array.from(s).map(c => c.charCodeAt(0));
                            }
                            function codesToStr(arr) {
                                return arr.map(c => String.fromCharCode(c)).join('');
                            }

                            // Encrypt message with RSA
                            function encryptRSA() {
                                let text = document.getElementById('rsaPlainText').value;
                                let p = parseInt(document.getElementById('rsaP').value, 10);
                                let q = parseInt(document.getElementById('rsaQ').value, 10);
                                if (!(isPrime(p) && isPrime(q) && p !== q)) {
                                    alert('Both p and q must be different valid prime numbers.');
                                    return;
                                }
                                let n = p * q;
                                let phi = (p-1)*(q-1);
                                let e = [3,5,17,257].find(candidate => gcd(candidate, phi) === 1);
                                if (!e) { alert("No valid public exponent was found."); return; }
                                let codes = strToCodes(text);
                                if (codes.some(code => code >= n)) {
                                    alert("n=p*q must be greater than the maximum character value.");
                                    return;
                                }
                                let encrypted = codes.map(c => modpow(c, e, n));
                                document.getElementById('rsaCipherText').value = encrypted.join(' ');
                                document.getElementById('rsaDecryptedText').value = ""; // clear previous
                            }

                            // Break/decrypt RSA (factor n, find e, get d, decrypt)
                            function breakRSA() {
                                let p = parseInt(document.getElementById('rsaP').value, 10);
                                let q = parseInt(document.getElementById('rsaQ').value, 10);
                                let n = p * q;
                                let cipherText = document.getElementById('rsaCipherText').value.trim();
                                if (!cipherText) {
                                    alert('Ciphertext not present.');
                                    return;
                                }
                                let encryptedBlocks = cipherText.split(/\s+/).map(Number);
                                // Try to factor n: check all divisors up to sqrt(n)
                                let found_p = null, found_q = null;
                                for (let i = 2; i*i <= n; ++i) {
                                    if (n % i === 0 && isPrime(i) && isPrime(n/i)) {
                                        found_p = i; found_q = n/i;
                                        break;
                                    }
                                }
                                if (!found_p) {
                                    alert('I was unable to factor n.');
                                    return;
                                }
                                let phi = (found_p-1)*(found_q-1);
                                // Try small common exponents e
                                let possibleEs = [3,5,17,257];
                                let recoveredE = possibleEs.find(e => gcd(e, phi) === 1);
                                if (!recoveredE) {
                                    alert('I was unable to determine the exponent e.');
                                    return;
                                }
                                let d = modinv(recoveredE, phi);
                                if (d === null) {
                                    alert('I couldn\'t calculate d.');
                                    return;
                                }
                                let decryptedCodes = encryptedBlocks.map(block => modpow(block, d, n));
                                let decryptedText = codesToStr(decryptedCodes);
                                document.getElementById('rsaDecryptedText').value = decryptedText;
                            }
